#include <Arduino.h>

#define UP_PIN  4
#define DOWN_PIN  5
#define LEFT_PIN  6
#define RIGHT_PIN  7
#define CENTER_PIN  10



int lastUpButton = LOW;
int lastDownButton = LOW;
int lastLeftButton = LOW;
int lastRightButton = LOW;
int lastCenterButton = LOW;

void setColor();

void setup() {

  pinMode(UP_PIN, INPUT_PULLDOWN);
  pinMode(DOWN_PIN, INPUT_PULLDOWN);
  pinMode(LEFT_PIN, INPUT_PULLDOWN);
  pinMode(RIGHT_PIN, INPUT_PULLDOWN);
  pinMode(CENTER_PIN, INPUT_PULLDOWN);


  Serial.begin(115200);
}

void loop() {
  int upButtonValue = digitalRead(UP_PIN);
  int downButtonValue = digitalRead(DOWN_PIN);
  int leftButtonValue = digitalRead(LEFT_PIN);
  int rightButtonValue = digitalRead(RIGHT_PIN);
  int centerButtonValue = digitalRead(CENTER_PIN);


  if (( upButtonValue == HIGH && leftButtonValue == HIGH) && (lastUpButton == LOW && lastLeftButton == LOW)) {
    Serial.println("up - left");
    delay(200);
  } if (( upButtonValue == HIGH && rightButtonValue == HIGH) && (lastUpButton == LOW && lastRightButton == LOW)) {
    Serial.println("up - right");
  } if ((downButtonValue == HIGH && leftButtonValue == HIGH) && (lastDownButton == LOW && lastLeftButton == LOW)) {
    Serial.println("down - left");
    delay(200);
  } if ((downButtonValue == HIGH && rightButtonValue == HIGH) && (lastDownButton == LOW && lastRightButton == LOW)) {
    Serial.println("down - right");
    delay(200);
  }

  if (upButtonValue == HIGH && lastUpButton == LOW) {
    Serial.println("up");
    delay(200);
  } if (downButtonValue == HIGH && lastDownButton == LOW) {
    Serial.println("down");
    delay(200);
  } if (leftButtonValue == HIGH && lastLeftButton == LOW) {
    Serial.println("left");
    delay(200);
  } if (rightButtonValue == HIGH && lastRightButton == LOW) {
    Serial.println("right");
    delay(200);
  } if (centerButtonValue == HIGH && lastCenterButton == LOW) {
    Serial.println("center");
    delay(200);
  }

  


  
  lastUpButton = upButtonValue;
  lastDownButton = downButtonValue;
  lastLeftButton = leftButtonValue;
  lastRightButton = rightButtonValue;
  lastCenterButton = centerButtonValue;
}
